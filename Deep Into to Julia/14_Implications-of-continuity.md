````julia
function bisection(f, a, b)

  f(a) == 0 && return(a)
  f(b) == 0 && return(b)

  f(a) * f(b) > 0 && error("[a, b] is not in a bracketing interval.")

  tol = 1e-14 # small epsilon
  c = a/2 + b/2

  while abs(b-a) > tol

    f(c) == 0 && return(c)

    if f(a) * f(c) < 0
      a, b = a, c
    else
      a, b = c, b
    end

    c = a/2 + b/2

  end

  c
end
````
